#### 插入排序

```java
public static void insertSort(int[] data) {
    for (int i = 1; i < data.length; i++) {
        for (int j = i; j > 0; j--) {
            if (data[j] < data[j - 1]) {
                int temp = data[j - 1];
                data[j - 1] = data[j];
                data[j] = temp;
            }
        }
    }
}
```

#### 选择排序

```java
public static void selectSort(int[] data) {
    for (int i = 0; i < data.length; i++) {
        int min = i;
        for (int j = i + 1; j < data.length; j++) {
            if (data[j] < data[min]) {
                min = j;
            }
        }
        if (i != min) {
            int temp = data[i];
            data[i] = data[min];
            data[min] = temp;
        }
    }
}
```

#### 冒泡排序

```java
public static void selectSort(int[] data) {
    for (int i = data.length - 1; i > 0; i--) {
        for (int j = 0; j < i; j++) {
            if (data[j] > data[j + 1]) {
                int temp = data[j];
                data[j] = data[j + 1];
                data[j + 1] = temp;
            }
        }
    }
}
```
