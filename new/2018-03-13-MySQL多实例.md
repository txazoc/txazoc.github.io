```shell
mkdir /usr/local/mysql/data_3307
```

```
bin/mysql_install_db --user=mysql --basedir=/usr/local/mysql --datadir=/usr/local/mysql/data_3307
```

```
cp /usr/local/mysql/my.cnf /usr/local/mysql/my_3307.cnf
```

```
[mysqld]
user = mysql
port = 3307
socket = /tmp/mysql_3307.sock
basedir = /usr/local/mysql
datadir = /usr/local/mysql/data_3307
log-error = /usr/local/mysql/data_3307/txazo.local.err
pid-file = /usr/local/mysql/data_3307/txazo.local.pid
```

```
bin/mysqld_safe --defaults-file=/usr/local/mysql/my_3307.cnf --user=mysql &
```
